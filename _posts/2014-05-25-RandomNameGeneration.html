                    class Squeue {
                        //S1 will hold elements in "limbo"
                        //S2 will then pop elements off of S1 to serve as the queue
                        Stack s1, s2;

                            public Squeue() {
                            s1 = new Stack();
                            s2 = new Stack();
                            }

                            public void enqueue(E item) {
                            s1.push(item);
                            }

                            @TEST
                            public E dequeue() {
                            //If S2 is empty, then we need to empty S1
                            if(s2.isEmpty()) {
                            //If S1 is also empty, there are no elements to dequeue
                            if(s1.isEmpty()) {
                            return null;
                            }

                            while(!s1.isEmpty()) {
                            s2.push(s1.pop());
                            }
                            }

                            return s2.pop();
                            }

                            public boolean isEmpty() {
                            return s1.isEmpty() && s2.isEmpty();
                            }

                            public String toString() {
                            String val = "S1: "+s1+"  S2: "+s2;
                            return val;
                            }
